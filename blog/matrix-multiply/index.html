    while (ch>='0'&&ch<='9') ret=ret*10+ch-'0',ch=getchar();

struct Matrix{
    int a[maxn][maxn];
    void ReadMatrix(){
        memset(a,0,sizeof(a));
        for (int i=0;i<n;i++)
        for (int j=0;j<n;j++) a[i][j]=read();
